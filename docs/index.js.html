function counterFactory(min, max, step = 1) {
    return function* counter(options = {}) {
        if (istype_1.isInteger(options)) {
            options = { initial: options };
        }
        const { initial = min } = options;
        let value = initial;
        while (1) {
            if (value > max) {
                value = min;
            }
            yield value;
            value += step;
        }

        return;
    };
}
